import 'package:flutter/material.dart'; import 'package:flutter_blue_plus/flutter_blue_plus.dart'; import 'package:nfc_manager/nfc_manager.dart'; import 'package:wifi_iot/wifi_iot.dart'; import 'package:permission_handler/permission_handler.dart'; void main() { runApp(const FlipperHackerApp()); } class FlipperHackerApp extends StatelessWidget { const FlipperHackerApp({super.key}); @override Widget build(BuildContext context) { return MaterialApp( title: 'Flipper Hacker', theme: ThemeData.dark().copyWith( primaryColor: Colors.deepPurple, scaffoldBackgroundColor: const Color(0xFF0A0A0A), ), home: const HomeScreen(), debugShowCheckedModeBanner: false, ); } } class HomeScreen extends StatefulWidget { const HomeScreen({super.key}); @override State createState() => _HomeScreenState(); } class _HomeScreenState extends State { String status = "Pronto para hackear 🚀"; @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar( title: const Text('🔧 Flipper Hacker', style: TextStyle(fontWeight: FontWeight.bold)), backgroundColor: Colors.deepPurple, ), body: Padding( padding: const EdgeInsets.all(16.0), child: Column( children: [ Container( padding: const EdgeInsets.all(20), decoration: BoxDecoration( color: Colors.deepPurple.withOpacity(0.2), borderRadius: BorderRadius.circular(16), ), child: Text(status, style: const TextStyle(fontSize: 18)), ), const SizedBox(height: 30), _buildMenuItem(Icons.bluetooth, "Bluetooth Scanner", () => scanBluetooth()), _buildMenuItem(Icons.nfc, "NFC / RFID", () => readNFC()), _buildMenuItem(Icons.wifi, "WiFi Scanner", () => scanWifi()), _buildMenuItem(Icons.settings_remote, "Infravermelho (IR)", () => irDemo()), ], ), ), ); } Widget _buildMenuItem(IconData icon, String title, VoidCallback onTap) { return Card( color: const Color(0xFF1E1E1E), margin: const EdgeInsets.symmetric(vertical: 8), child: ListTile( leading: Icon(icon, color: Colors.deepPurple, size: 32), title: Text(title, style: const TextStyle(fontSize: 18)), trailing: const Icon(Icons.arrow_forward_ios), onTap: onTap, ), ); } Future scanBluetooth() async { setState(() => status = "Escaneando Bluetooth..."); try { await FlutterBluePlus.startScan(timeout: const Duration(seconds: 5)); setState(() => status = "Dispositivos próximos detectados! (verifique logs)"); } catch (e) { setState(() => status = "Erro no Bluetooth: $e"); } } Future readNFC() async { setState(() => status = "Aproxime um cartão/tag NFC..."); bool isAvailable = await NfcManager.instance.isAvailable(); if (!isAvailable) { setState(() => status = "NFC não disponível neste dispositivo"); return; } NfcManager.instance.startSession( onDiscovered: (NfcTag tag) async { setState(() => status = "NFC detectado!\nID: ${tag.id}"); await NfcManager.instance.stopSession(); }, ); } Future scanWifi() async { setState(() => status = "Escaneando WiFi..."); await Permission.location.request(); List networks = await WifiIotPlugin.listNetworks(); setState(() => status = "${networks.length} redes WiFi encontradas!"); } void irDemo() { setState(() => status = "IR não nativo na maioria dos celulares modernos.\nUse hardware externo se possível."); } }
